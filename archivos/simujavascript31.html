Una función condicional compleja

En este ejemplo la función funcioncomparar() tiene una sola función condicional que incorpora cuatro condiciones, dos de ellas pueden cumplirse por separado, pero han de cumplirse a la vez con las otras dos.

El conjunto de todas las condiciones es este: if((variableuno>10)&&(variabledos>10)&&((variableuno==variabledos*3)||(variabledos==variableuno*3))).

Ahora puedes elaborar una función condicional diferente, con otras condiciones enlazadas entre si.

Haz todas las modificaciones que creas oportunas, para familiarizarte con el código.

var variableuno = 0;

var variabledos = 0;

function funcionvariableuno(indiceuno){

if(indiceuno==1){
variableuno = variableuno - 1;
}

if(indiceuno==2){
variableuno = variableuno + 1;
}

document.getElementById("muestravariableuno").innerHTML = variableuno;

funcioncomparar();

}

function funcionvariabledos(indicedos){

if(indicedos==1){
variabledos = variabledos - 1;
}

if(indicedos==2){
variabledos = variabledos + 1;
}

document.getElementById("muestravariabledos").innerHTML = variabledos;

funcioncomparar();

}

function funcioncomparar(){

var mensaje ="";

if((variableuno>10)&&(variabledos>10)&&((variableuno==variabledos*3)||(variabledos==variableuno*3))){
mensaje = "Las dos variables son mayores de 10 y una de ellas es el triple de la otra.;"
}

document.getElementById("muestramensaje").innerHTML = mensaje;

}
