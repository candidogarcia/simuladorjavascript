Que salten los delfines

En este ejemplo, similar al de las ruedas dentadas, dentro del canvas se mueven tres delfines que parece que salten, aunque en realidad giran. La imagen del delfín es png para permitir transparencia y tiene estas dimensiones.

La altura del canvas es más pequeña para no mostrar parte del recorrido del delfín. La orden context.drawImage(cuartaimagen, 200, 20); situa la imagen fija del Sol en las coordenadas deseadas.

Haz todas las modificaciones que creas oportunas en la animación, para familiarizarte con el código.

<script type="text/javascript">

var contador = 0;
primeraimagen = new Image();
primeraimagen.src = '';
segundaimagen = new Image();
segundaimagen.src = '';
terceraimagen = new Image();
terceraimagen.src = '';
cuartaimagen = new Image();
cuartaimagen.src = '';
radian = Math.PI/180;
var canvas = document.getElementById("canvas");
var context = canvas.getContext('2d');
context.fillStyle="white";
context.fillRect(0, 0, canvas.width, canvas.height);

setInterval(loop, );

function loop() {

context.clearRect(0,0,canvas.width, canvas.height);
context.fillStyle="";
context.fillRect(0, 0, canvas.width, canvas.height);
context.strokeRect(0, 0, canvas.width, canvas.height);
context.drawImage(cuartaimagen, , );
rotarimagen(primeraimagen, ,, /);
rotarimagen(segundaimagen, ,, /);
rotarimagen(terceraimagen, ,, /);
contador+=2;

}

function rotarimagen(imagen, x, y, angulo) {

context.save();
context.translate(x, y);
context.rotate(angulo * radian);
context.drawImage(imagen, -(imagen.width/2), -(imagen.height/2));
context.restore();

}
