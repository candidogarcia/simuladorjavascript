Un satélite artificial pasa delante del Sol

En este ejemplo, de nuevo, una imagen se desplaza y gira. El satélite artificial se desplaza en diagonal, desde la esquina superior izquierda hasta la inferior derecha. A la vez, va aumentando su tamaño, proporcionando la sensación de que se acerca a nosotros. Una vez ha llegado a la esquina inferior se vuelve a dar valor a las diferentes variables, para que vuelva a aparecer por la esquina superior izquierda.

Haz todas las modificaciones que creas oportunas en la animación, para familiarizarte con el código.

<script type="text/javascript">

var contador = 0;
var primeraimagenx = ;
var primeraimageny = ;
var segundaimagenx = 0;
var segundaimageny = 0;
var anchoimagen = 137;
var altoimagen = 137;
var incrementoimagen = 0.1
primeraimagen = new Image();
primeraimagen.src = '';
segundaimagen = new Image();
segundaimagen.src = '';
radian = Math.PI/180;
var canvas = document.getElementById("canvas");
var context = canvas.getContext('2d');
context.fillStyle="#FFFFFF";
context.fillRect(0, 0, canvas.width, canvas.height);

setInterval(loop, );

function loop() {

context.clearRect(0,0,canvas.width, canvas.height);
context.fillStyle="";
context.fillRect(0, 0, canvas.width, canvas.height);
context.strokeRect(0, 0, canvas.width, canvas.height);
context.drawImage(primeraimagen, primeraimagenx, primeraimageny);
rotarimagen(segundaimagen, segundaimagenx, segundaimageny, contador/, anchoimagen, altoimagen);

anchoimagen = anchoimagen + incrementoimagen;
altoimagen = altoimagen + incrementoimagen;

segundaimagenx = segundaimagenx + 0.1;
segundaimageny = segundaimageny + 0.1;

if(segundaimagenx>500){
segundaimagenx = 0;
segundaimageny = 0;
anchoimagen = 137;
altoimagen = 137;
}

contador+=2;

}

function rotarimagen(imagen, x, y, angulo, anchoimagen, altoimagen) {

context.save();
context.translate(x, y);
context.rotate(angulo * radian);
context.drawImage(imagen, -(anchoimagen/2), -(altoimagen/2), anchoimagen, altoimagen);
context.restore();

}
