Primera calculadora

En este ejemplo tenemos una calculadora con un teclado con las 10 cifras, las cuatro operaciones básicas y el igual para obtener el resultado.

Las variables numeroprimero = "" y numerosegundo = "" son cadenas de texto en donde vamos introduciendo el valor de los dos números con los que vamos a operar. Para obtener el resultado final hemos de transformar estas cadenas alfanuméricas en números mediante la orden parseInt(numeroprimero);.

Cuando se hace clic sobre una tecla numérica se llama a la función introducirnumero(numero) en donde se va sumando a la cadena numeroprimero el valor que corresponda. Cuando se aprieta la tecla de operación se llama a la función operar(indice), en ella se cambia el valor de la variable ordennumero a uno, para que cuando se vuelva a introducir un número este se guarde en la cadena numerosegundo. Además se asigna valor a la variable operación, dependiendo de la operación a realizar. Por fin, cuando se pulsa la tecla igual se llama a la función calcular(). En ella se tranforman en números las dos cadenas numeroprimero y numerosegundo y, a continuación, mediante cuatro funciones condicionales se asigna un valor a la variable resultado y se coloca en el párrafo identificado como resultados. Para acabar se borran las cadenas numeroprimero y numerosegundo y se cambia el valor de la variable ordennumero a cero, para que se pueda iniciar otro cálculo.

Haz todas las modificaciones que creas oportunas, para familiarizarte con el código.

<script type="text/javascript">

var numeroprimero = "";

var numerosegundo = "";

var numerofinal = 0;

var operacion = 0;

var ordennumero = 0;

function introducirnumero(numero){

if(ordennumero==0){
numeroprimero = numeroprimero + numero;
document.getElementById("primernumero").innerHTML = numeroprimero;
}

if(ordennumero==1){
numerosegundo = numerosegundo + numero;
document.getElementById("segundonumero").innerHTML = numerosegundo;
}

}

function operar(indice){

if(indice==1){
operacion = 1;
}

if(indice==2){
operacion = 2;
}

if(indice==3){
operacion = 3;
}

if(indice==4){
operacion = 4;
}

ordennumero = 1;

}

function calcular(){

var numerouno = parseInt(numeroprimero);
var numerodos = parseInt(numerosegundo);

if(operacion==1){
resultado = numerouno + numerodos;
}

if(operacion==2){
resultado = numerouno - numerodos;
}

if(operacion==3){
resultado = numerouno * numerodos;
}

if(operacion==4){
resultado = numerouno / numerodos;
}

document.getElementById("resultados").innerHTML = resultado;

numeroprimero = "";

numerosegundo = "";

ordennumero = 0;

}

</script>
